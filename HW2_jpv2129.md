``` r
unemployment_df = read_csv("data/unemployment.csv") |> 
  janitor::clean_names() |> 
  pivot_longer(
    cols = jan:dec,
    names_to = "month_abbr",
    values_to = "unemployment_rate"
  ) |> 
  # Convert abbreviations using month.name and month.abb constants
  mutate(month = month.name[match(month_abbr, tolower(month.abb))]) |> 
  select(-month_abbr) |> 
  rename(year = year) |> 
  arrange(year, month)
```
